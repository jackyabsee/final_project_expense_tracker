-------command to create table---------
CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  account VARCHAR(20),
  email VARCHAR(50),
  password VARCHAR,
  username VARCHAR(10)
);

CREATE TABLE spending (
  id SERIAL PRIMARY KEY,
  type VARCHAR,
  price VARCHAR,
  date DATE,
  remark VARCHAR,
  user_id INTEGER NOT NULL REFERENCES users(id)
);

CREATE TABLE asset (
  id SERIAL PRIMARY KEY,
  category VARCHAR(10),
  organization VARCHAR(12),
  type VARCHAR(12),
  value VARCHAR(10),
  interest_rate VARCHAR,
  remark VARCHAR,
  user_id INTEGER NOT NULL REFERENCES users(id)
);

CREATE TABLE content (
  id SERIAL PRIMARY KEY,
  title VARCHAR(40),
  url VARCHAR
);
